# Thanos-Snap-Simulator
This Python program simulates the famous snap of Thanos from Avengers: Infinity War. It takes a list of characters and randomly removes half of them, just like the snap that erased half the universe. Each time the code runs, different survivors remain, showing randomness in action.



import random import math def thanos_snap(population): """ Randomly eliminates half of the items in a given list. Args: population: A list of items. Returns: A new list containing the surviving half of the items. """ initial_length = len(population) # Calculate the number of items to eliminate (half, rounded up) items_to_eliminate_count = math.ceil(initial_length / 2) # Create a copy to work with, to avoid modifying the original list directly survivors = list(population) # Randomly select and remove items until half are eliminated for _ in range(items_to_eliminate_count): if not survivors: break item_to_remove = random.choice(survivors) survivors.remove(item_to_remove) return survivors # --- Example Usage --- # Create a sample list of names universe_inhabitants = ["Iron Man", "Captain America", "Thor", "Hulk", "Black Widow", "Hawkeye", "War Machine", "Falcon", "Scarlet Witch", "Vision"] print(f"Initial population: {len(universe_inhabitants)} inhabitants") print(f"Initial list: {universe_inhabitants}\n") # Perform the snap surviving_inhabitants = thanos_snap(universe_inhabitants) print(f"Remaining population after snap: {len(surviving_inhabitants)} inhabitants") print(f"Survivors: {surviving_inhabitants}")
